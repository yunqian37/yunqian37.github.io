    let obj1 = {
      name: 'zhangSan',

        city: 'changZhou'

    }
    let obj2 = deepClone(obj1)
    obj2.name = 'liSi'

    console.log('obj1 =>', obj1)

    function deepClone( obj = {} ) {
      if (typeof(obj) !== 'object' || obj == null) {
        return obj
      }
      let result
      if (obj instanceof Array) {
        result = []
      } else {
        result = {}
      }

      for (let key in obj) {
        if (obj.hasOwnProperty(key)) {
          result[key] = deepClone(key)
        }
      }
      return result
    }
  </script>
